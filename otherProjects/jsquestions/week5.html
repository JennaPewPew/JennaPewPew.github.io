Write a code that counts the number of Capital B's in a sentence. 
 
              function countBs(bString) {
                var count = 0;
                for (var i = 0; i < bString.length; i++) {
                  if (bString.charAt(i) == 'B') {
                      count = (count + 1);
                  }
                }
                  return count;
              }


              $("#q4button").on("click", function() {

                var userInput = prompt ("Please enter a sentence");
                var x = countBs(userInput);

                $("#q4result").html(x);
              });
